50 /**
51  * Extend default arguments. Allows you to pass associative arrays to template functions with defaults,
52  * and merge in with passed in args.
53  * 
54  * @package d7
55  * @subpackage utility
56  * 
57  * @param array $args   Associative array of arguments to merge
58  * 
59  * @link http://gabrieleromanato.name/php-using-associative-arrays-to-handle-default-function-arguments/
60  * 
61  */
62 function extend_args($args, $defaults = array()) {
63 
64     if ( is_object( $args ) ) {
65         $result = get_object_vars( $args );
66     } elseif ( is_array( $args ) ) {
67         $result =& $args;
68     }
69 
70     if ( is_array( $defaults ) ) {
71         return array_merge( $defaults, $result );
72     }
73 
74     return $result;
75 }
